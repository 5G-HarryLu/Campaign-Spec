Why Campaign Does Not Use a Distributed Lock

雖然上方列出分散式鎖為可複用模式，Campaign (FRB) 服務刻意不採用，原因如下：

Message loss risk during rolling deploys — if a pod holding the lock gets terminated, the message is lost with no retry mechanism
Redis Pub/Sub is fire-and-forget — messages can't be replayed, so we broadcast to all pods and rely on redundant execution instead
Existing safeguards are sufficient — OceanBase SELECT FOR UPDATE, idempotency check (ExistsAsync), unique index (uk_campaign_operator_member), and Polly retry together prevent duplicates without a distributed lock
Added complexity for little benefit — lock acquisition/release, lock expiry tuning, and failure handling add operational risk; the current approach is simpler and achieves the same correctness

public interface IScoreCalculator
{

public class MaxMultiplierCalculator : IScoreCalculator { ... }
public class AccumulatedMultiplierCalculator : IScoreCalculator { ... }
public class TotalWinCalculator : IScoreCalculator { ... }
public class TurnoverCalculator : IScoreCalculator { ... }

public class ScoreCalculatorFactory
{
    public IScoreCalculator Create(ScoringMode mode) => mode switch
    {
        ScoringMode.MaxMultiplier => new MaxMultiplierCalculator(),
        ScoringMode.AccumulatedMultiplier => new AccumulatedMultiplierCalculator(),
        ScoringMode.TotalWin => new TotalWinCalculator(),
        ScoringMode.Turnover => new TurnoverCalculator(),
        _ => throw new ArgumentOutOfRangeException()
    };
}

public class RewardCalculator
{
    public List<PlayerReward> Calculate(
        List<RankEntry> finalRanking,
        List<PrizePoolTier> prizeConfig)
    {
        var rewards = new List<PlayerReward>();
        foreach (var tier in prizeConfig)
        {
            var playersInTier = finalRanking
                .Where(r => r.Rank >= tier.RankFrom
                          && r.Rank <= tier.RankTo);
            foreach (var player in playersInTier)
            {
                rewards.Add(new PlayerReward
                {
                    PlayerId = player.MemberId,
                    Rank = player.Rank,
                    Amount = tier.Prize,
                    Currency = tournament.Currency,
                    Status = RewardStatus.Pending
                });
            }
        }
        return rewards;
    }
}
